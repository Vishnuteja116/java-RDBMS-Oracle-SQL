inner join
select ename,e.eid,d.eid,dname 
from emp541 e,dept541 d
where e.eid=d.eid;
///
left outer join
select ename,dname,e.eid,d.eid
from emp541 e left outer join dept541 d
on e.eid=d.eid;
///
right outer join
select ename,dname,e.eid,d.eid
from emp541 e right outer join dept541 d
on e.eid=d.eid;
///
full outer join
select ename,dname,e.eid,d.eid
from emp541 e full outer join dept541 d
on e.eid=d.eid;
///
cross join
cross product of two tables
select ename,dname,e.eid,d.eid
from emp541 e cross join dept541 d
on e.eid=d.eid;

self join

hierarchical query
